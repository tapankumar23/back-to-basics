Dijkstra & Bellman-Ford

// Dijkstra's Algorithm — O((V+E) log V)
int[] dijkstra(int n, List<int[]>[] adj, int src) {
    int[] d = new int[n];
    Arrays.fill(d, Integer.MAX_VALUE);
    d[src] = 0;
    // min-heap: [dist, node]
    PriorityQueue<int[]> pq = new PriorityQueue<>((a, b) -> a[0] - b[0]);
    pq.offer(new int[]{0, src});
    while (!pq.isEmpty()) {
        int[] c = pq.poll();
        int dist = c[0], u = c[1];
        if (dist > d[u]) continue; // stale entry
        for (int[] e : adj[u]) {
            int v = e[0], w = e[1];
            if (d[u] + w < d[v]) {
                d[v] = d[u] + w;
                pq.offer(new int[]{d[v], v});
            }
        }
    }
    return d;
}

// Bellman-Ford — O(V·E), detects negative cycles
int[] bellmanFord(int n, int[][] edges, int src) {
    int[] d = new int[n];
    Arrays.fill(d, Integer.MAX_VALUE);
    d[src] = 0;
    for (int i = 0; i < n - 1; i++) {
        for (int[] e : edges) {
            if (d[e[0]] != Integer.MAX_VALUE && d[e[0]] + e[2] < d[e[1]])
                d[e[1]] = d[e[0]] + e[2];
        }
    }
    // Nth pass: if anything relaxes, negative cycle exists
    for (int[] e : edges)
        if (d[e[0]] != Integer.MAX_VALUE && d[e[0]] + e[2] < d[e[1]])
            return new int[0]; // negative cycle detected
    return d;
}

// LC 787: Cheapest Flights Within K Stops
int findCheapestPrice(int n, int[][] flights, int src, int dst, int k) {
    int[] prices = new int[n];
    Arrays.fill(prices, Integer.MAX_VALUE);
    prices[src] = 0;
    for (int i = 0; i <= k; i++) {
        int[] tmp = Arrays.copyOf(prices, n);
        for (int[] f : flights)
            if (prices[f[0]] != Integer.MAX_VALUE)
                tmp[f[1]] = Math.min(tmp[f[1]], prices[f[0]] + f[2]);
        prices = tmp;
    }
    return prices[dst] == Integer.MAX_VALUE ? -1 : prices[dst];
}
